\section{Geodesic \& Boltzmann Equations for photons}
%=========================================================
%------------------------------------------------------------

We report the newtonian-gauge geodesic equations \eqref{eq:geodesic_equation_covariant_momentum_newtonian}-\eqref{eq:geodesic_equation_physical_momentum_newtonian} fot photons, which simplify since $E=p$,
\begin{align}\label{eq:geodesic_equations_photon_newtonian}
\begin{aligned}
    \frac{d P^0}{d\lambda}
    &= -\frac{E^2}{a^2}\left(\mathcal{H}(1-2\psi)+\dot\psi\right)-2\frac{Ep}{a^2}\hat{p}^j\partial_j\psi-\frac{p^2}{a^2}\left(\mathcal{H}(1-2\psi)+\dot\phi\right),
    \\
    \frac{dP^k}{d\lambda}&=-\frac{E^2}{a^2}\partial_k\psi-\frac{Ep}{a^2}\hat{p}^k\,2\left(\mathcal{H}(1-\psi+\phi)-\dot\phi\right)-\frac{p^2}{a^2}\left(\partial_k\phi-2\hat{p}^k\hat{p}^j\partial_j\phi\right).
    \\
    \frac{d E}{d\ct}&=-\frac{E^2}{p}(\mathcal{H}-\dot\phi)-p\hat{p}^j\partial_j\psi = -E\mathcal{H}+E\frac{d\phi}{d\tau}-p\hat{p}^j(\partial_j\phi+\partial_j\psi),
    \\
    \frac{d\hat{p}^i}{d\ct}&=\big(\hat{p}^i\hat{p}^j-\delta^{ij}\big)\left(\frac{E}{p}\partial_j\psi+\frac{p}{E}\partial_j\phi\right).
\end{aligned}
\end{align}
The last equation implies that to zeroth order in metric perturbations, photons travel along straight lines, while at first order they are deflected by the gravitational potentials $\psi$ and $\phi$.
The third equation encodes the change in energy along the trajectory.
It is convenient to rewrite it trading partial with total time derivatives $\frac{\partial \phi}{\partial \tau}=\frac{d\phi}{d\tau}-\frac{\partial \phi}{\partial x^j}\frac{dx^j}{d\tau}$ to get
{\small
\begin{align}
    \frac{d \log E}{d\tau}&= -\mathcal{H}+\frac{d\phi}{d\tau}-\hat{p}^j(\partial_j\phi+\partial_j\psi).
\end{align}}
To order zero it encode the expected redshift behavior $E=p=\propto\, a^{-1}$.
The perturbed part implies energy is red-blueshifted by the gravitational potentials via the istantaneous and integrated Sachs-Wolf effect.
Since the change in $\hat{p}^j$ along the trajectory is already first order, we can integrate it to get
{\small
\begin{align}
    \log \Big(\frac{E_f}{E_i}\Big) &= \int_{\tau_i}^{\tau_f} d\tau\,\, -\mathcal{H} + \frac{d\phi}{d\tau} - \hat{p}^j(\partial_j\phi+\partial_j\psi) \, d\tau\\
    &= \log \Big(\frac{a_i}{a_f}\Big) + \underbrace{\big(\phi(\tau_f,x_f)-\phi(\tau_i,x_i)\big)}_{\text{instantaneous SW (aka gravitational redshift)}} - \int_{\tau_i}^{\tau_f} \underbrace{d\tau\,\, \hat{p}^j(\partial_j\phi+\partial_j\psi)}_{\text{integrated Sachs-Wolfe effect}}.
\end{align}}
Writing $E = \bar{E}(1+\Theta)$ with $\bar{E}\propto \, a^{-1}$ so that $\bar{E}_ia_i=\bar{E}_fa_f$ to remove the background redshift, we get the expected Sachs-Wolfe formula for the observed temperature anisotropies
{\small
\begin{align}
    \Theta_f(\hat{p})-\Theta_i(\hat{p})\simeq \log\left(\frac{1+\Theta_f(\hat{p})}{1+\Theta_i(\hat{p})}\right) 
    = \phi(\tau_f,x_f)-\phi(\tau_i,x_i) - \int_{\tau_i}^{\tau_f} d\tau\,\, \hat{p}^j(\partial_j\phi+\partial_j\psi).
\end{align}}


We report the Boltzmann equation \eqref{eq:Boltzmann_LHS_exact_newton_gauge} for photons becomes (second order in metric perturbations, exact otherwise)
{\small
\begin{align}\label{eq:Boltzmann_LHS_exact_newton_gauge_photon}
\begin{aligned}
    C[f]&=\frac{d f}{d \lambda}
    =
    %
    P^{0}\left[\frac{\partial f}{\partial \tau}+\frac{\partial f}{\partial x^{i}} \frac{P^{i}}{P^{0}}+\frac{\partial f}{\partial E} \frac{1}{P^{0}} \frac{d E}{d \lambda}+\frac{\partial f}{\partial \hat{p}^{i}} \frac{1}{P^{0}} \frac{d \hat{p}^{i}}{d \lambda}\right]
    \\
    &=\frac{E}{a}\!\left[\frac{\partial f}{\partial \ct}(1-\psi)
    \!+\!\left(\hat{p}^j\partial_jf\right) \frac{p}{E}(1\!+\!\phi)
    -\frac{\partial f}{\partial E}\frac{p^2}{E}\left(\mathcal{H}(1\!-\!\psi)-\dot\phi\right)
    -\frac{\partial f}{\partial \bar{p}^j}\partial_j\psi \,E
    \!+\!\left(\frac{\partial f}{\partial p} \hat{p}^j-\frac{\partial f}{\partial \bar{p}^j}\right)\partial_j\phi\,\frac{p^2}{E}\right].
\end{aligned}
\end{align}}

Photons retain local equilibrium with a blackbody distribution at all time\todotag{Discuss}, simply with a time and momentum dependent temperature, 
{\small
\begin{equation}\label{eq:local_temp_photons}
    f(\ct, x^i, E, \hat{p}^i) = \frac{1}{\exp\left(\frac{E}{T(\ct, x^i)}\right) - 1}, \quad T=\bar{T}(\tau)(1+\Theta(\tau, x^i, p, \hat{p}^i))=\bar{T}+\delta T.
\end{equation}}
More generally, assuming the full distribution is still a function $f(\nicefrac{E}{T})$ of the ratio $\nicefrac{E}{T}$ to some \emph{local} temperature $T$, we Taylor expand around the background $\bar{f}(\tau,E)=\bar{f}(\nicefrac{E}{\bar{T}(\tau)})$ to get
{\small
\begin{align}
    f&= \bar{f}+\frac{\partial f}{\partial T}\delta T = \bar{f}-\frac{\partial \bar{f}}{\partial E}E\Theta = \bar{f}-\frac{d \bar{f}}{d (\nicefrac{E}{\bar{T}})} \frac{E}{\bar{T}}\Theta,\qquad\qquad \text{with}\quad \frac{d \bar{f}}{d (\nicefrac{E}{\bar{T}})}=\frac{\partial \bar{f}}{\partial E}_{|\tau}\bar{T}=-\frac{\partial \bar{f}}{\partial \bar{T}}_{|E}\frac{\bar{T}^2}{E}.
\end{align}}
In turn we compute the various derivatives appearing in the LHS of Boltzmann equation 
{\small
\begin{align}\label{eq:photon_distribution_expansion}
\begin{aligned}
    \frac{\partial f}{\partial \tau}&= -\frac{d\bar{f}}{d(\nicefrac{E}{\bar{T}})}\frac{E}{\bar{T}^2}\frac{d\bar{T}}{d\tau}+ \frac{d^2\bar{f}}{d(\nicefrac{E}{\bar{T}})^2} \frac{E^2}{\bar{T}^3}\frac{d\bar{T}}{d\tau}\Theta+\frac{d \bar{f}}{d(\nicefrac{E}{\bar{T}})}\frac{E}{\bar{T}^2}\frac{d\bar{T}}{d\tau}\Theta-\frac{d \bar{f}}{d (\nicefrac{E}{\bar{T}})} \frac{E}{\bar{T}}\frac{\partial \Theta}{\partial \tau}\\
    &= \frac{\partial \bar{f}}{\partial E}\frac{E}{\bar{T}}\frac{d\bar{T}}{d\tau}+\frac{\partial^2 \bar{f}}{\partial E^2}\frac{E^2}{\bar{T}}\frac{d\bar{T}}{d\tau}\Theta+\frac{\partial \bar{f}}{\partial E}\frac{E}{\bar{T}}\frac{d\bar{T}}{d\tau}\Theta-\frac{\partial \bar{f}}{\partial E}E\frac{\partial \Theta}{\partial \tau},
    \\
    \frac{\partial f}{\partial E}&=\frac{\partial \bar{f}}{\partial E}-\frac{\partial^2 \bar{f}}{\partial E^2}E\Theta-\frac{\partial \bar{f}}{\partial E}\Theta-\frac{\partial \bar{f}}{\partial E}E\frac{\partial \Theta}{\partial E},\\
    \frac{\partial f}{\partial \hat{p}^i}&=-E\frac{\partial \bar{f}}{\partial E}\frac{\partial \Theta}{\partial \hat{p}^i},\\
    \frac{\partial f}{\partial x^i}&=-E\frac{\partial \bar{f}}{\partial E}\frac{\partial \Theta}{\partial x^i}.
\end{aligned}
\end{align}}
Plugging \eqref{eq:photon_distribution_expansion} into the LHS of Boltzmann equation \eqref{eq:Boltzmann_LHS_exact_newton_gauge_photon} at zero order we get the redshift $T\propto a^{-1}$ of photon temperature
{\small
\begin{align}\label{eq:Boltzmann_LHS_photon_zero_order}
\begin{aligned}
    &\cancel{\tfrac{E}{a}E\partial_E\bar{f}}\bigg[\tfrac{1}{\bar{T}}\tfrac{d\bar{T}}{d\tau}+\mathcal{H}\bigg]=0 \quad \Rightarrow \quad \bar{T}\propto \frac{1}{a}.\\[3pt]
\end{aligned}
\end{align}}
At first order, replacing $\frac{\dot{T}}{T}=-\mathcal{H}$, we get
{\small
\begin{align}\label{eq:Boltzmann_LHS_photon_first_order_newtonian}
\begin{aligned}
    &-\tfrac{E}{a}E\partial_E\bar{f}\bigg[\frac{\partial\Theta}{\partial \tau}+\hat{p}^j\frac{\partial\Theta}{\partial x^j}-\mathcal{H}E\cancel{\frac{\partial \Theta}{\partial E}}+\hat{p}^j\frac{\partial\psi}{\partial x^j}-\frac{\partial \phi}{\partial\tau}\bigg] = C[f].
\end{aligned}
\end{align}}
At late times collisions with baryons are essentially low-energy Compton scatterings.
To good approximation, these scatterings do not change the energy of photons but only the direction of their momentum, and we can thus assume $\Theta=\Theta(\tau, x^i, \hat{p}^i)$, i.e. the temperature perturbation is independent of the photon energy.

\subsection{Passing to multipoles}
%====================================================================
We can now pass to Fourier space and decompose the angular dependence of $\Theta$ in multipoles to get a hierarchy of coupled ODEs for the multipole moments $\Theta_\ell$.
Passing to Fourier space in \eqref{eq:Boltzmann_LHS_photon_first_order_newtonian}, we get\todotag{Fix RHS}
{\small
\begin{align}\label{eq:Boltzmann_LHS_photon_first_order_newtonian_Fourier}
\begin{aligned}
    &\bigg[\frac{\partial\Theta}{\partial \tau}+i\vec{k}\cdot\hat{p}\,\Theta-\mathcal{H}E\cancel{\frac{\partial \Theta}{\partial E}}+i\hat{p}\cdot\vec{k}\,\psi-\frac{\partial \phi}{\partial\tau}\bigg] = - \frac{1}{\tfrac{E}{a}E\partial_E\bar{f}} C[f].
\end{aligned}
\end{align}}
The dependence on $\vec{k}$ only enter via the modulus $k$ and the angle $\mu:=\hat{k}\cdot\hat{p}$, so we can decompose the angular dependence in Legendre polynomials $P_\ell(\mu)$ as
{\small
\begin{align}
    \Theta(\tau, k, \mu) = \sum_{\ell=0}^\infty \frac{2\ell+1}{i^\ell} \Theta_\ell(\tau, k) P_\ell(\mu)\,,\quad
    \Theta_\ell(\tau, k) = \frac{i^\ell}{2}\int_{-1}^1 d\mu\, P_\ell(\mu) \Theta(\tau, k, \mu).
\end{align}}
The first three multipoles have a simple physical interpretation: up to numerical factors, $\Theta_0$ is the monopole i.e. the average temperature perturbation; $\Theta_1$ is the dipole i.e. the bulk velocity of the photon fluid; $\Theta_2$ is the quadrupole i.e. the anisotropic stress of the photon fluid.
Higher multipoles $\Theta_\ell$ with $\ell\geq 3$ encode higher-order angular correlations of the temperature perturbation, and are generated by free-streaming of photons after decoupling.

Recall Legendre multipoles are orthogonal, satisfy a recursion relation and the first 3 are
{\small
\begin{align}
    \int_{-1}^1 \!\!d\mu\, P_\ell(\mu) P_{\ell'}(\mu) = \tfrac{2}{2\ell+1} \delta_{\ell \ell'},\,\,
    \mu P_\ell(\mu) = \tfrac{\ell+1}{2\ell + 1} P_{\ell+1}(\mu) + \tfrac{\ell}{2\ell + 1} P_{\ell-1}(\mu),\,\,
    P_0(\mu) = 1, \,\, P_1(\mu) = \mu, \,\, P_2(\mu) = \frac{3\mu^2 - 1}{2}.
\end{align}}
Multiplying the equation by $\frac{i^\ell}{2}P_\ell(\mu)$ and integrating over $\mu$, using the orthogonality and recursion relations and the fact that $\psi$ and $\phi$ are independent of $\mu$, we get
{\small
\begin{align}
    \frac{\partial \Theta_\ell}{\partial \tau} + k\left(\frac{\ell+1}{2\ell + 1} \Theta_{\ell+1} - \frac{\ell}{2\ell + 1} \Theta_{\ell-1}\right) - \mathcal{H}E\cancel{\frac{\partial \Theta_\ell}{\partial E}} -\tfrac13 k\psi \delta_{\ell 1} - \frac{\partial \phi}{\partial\tau} \delta_{\ell 0} = - \frac{1}{\tfrac{E}{a}E\partial_E\bar{f}} C_\ell[f],
\end{align}}
The Boltzmann equation then becomes a hierarchy of coupled ODEs for multipole moments $\Theta_\ell$, which can be solved numerically to get the CMB power spectrum and other observables,\todotag{Fix RHS}
{\small\begin{align}
\begin{aligned}
    \frac{\partial \Theta_0}{\partial \log a} - E\cancel{\frac{\partial \Theta_0}{\partial E}}- \frac{\partial \phi}{\partial\log a}+ \frac{k}{\mathcal{H}} \Theta_1 &= - \frac{1}{\mathcal{H}}\tfrac{1}{\tfrac{E}{a}E\partial_E\bar{f}} C_0[f],\\
    \frac{\partial \Theta_1}{\partial \log a} -E\cancel{\frac{\partial \Theta_1}{\partial E}}+ \tfrac{2}{3} \frac{k}{\mathcal{H}} \Theta_2 -\tfrac13 \frac{k}{\mathcal{H}}\Theta_0- \tfrac{1}{3} \frac{k}{\mathcal{H}} \psi &= - \frac{1}{\mathcal{H}}\frac{1}{\tfrac{E}{a}E\partial_E\bar{f}} C_1[f],\\
    \frac{\partial \Theta_2}{\partial \log a} -E\cancel{\frac{\partial \Theta_2}{\partial E}}+ \tfrac{3}{5} \frac{k}{\mathcal{H}} \Theta_3 - \tfrac{2}{5} \frac{k}{\mathcal{H}} \Theta_1 &= - \frac{1}{\mathcal{H}}\frac{1}{\tfrac{E}{a}E\partial_E\bar{f}} C_2[f],\\
    \frac{\partial \Theta_\ell}{\partial \log a} -E\cancel{\frac{\partial \Theta_\ell}{\partial E}}+ \frac{k}{\mathcal{H}}\left(\tfrac{\ell+1}{2\ell + 1} \Theta_{\ell+1} - \tfrac{\ell}{2\ell + 1} \Theta_{\ell-1}\right) &= -\frac{1}{\mathcal{H}} \frac{1}{\tfrac{E}{a}E\partial_E\bar{f}} C_\ell[f], \quad \ell\geq 3.
\end{aligned}
\end{align}}
We passed to time variable $\log a$ to highlight dominant terms in the super(sub)-ubhorizon regime $k/\mathcal{H}\ll1$.
Note only the first 3 equations for $\Theta_0, \Theta_1, \Theta_2$ are sourced.
Metric potential enter only the first 2, while anisotropic stress enters the last one.

\section{The photon-baryon fluid}
%=========================================================
%------------------------------------------------------------

The relevant interactions for the photon-baryon fluid are
\begin{align}
    e^-+p \leftrightarrow e^-+p, \quad \text{(Coulomb scattering)}\\
    e^-+\gamma \leftrightarrow e^-+\gamma, \quad \text{(Thomson scattering:=low-energy Compton scattering)}\\
    p+\gamma \leftrightarrow p+\gamma, \quad \text{(Thomson scattering:=low-energy Compton scattering)}
\end{align}
In principle we would have 3 separate Boltzmann equations for electrons, protons and photons with corresponding collision terms
\begin{align}
\begin{aligned}
    \frac{d f_e}{d\lambda}&=C_{e\gamma}[f_e,f_\gamma]+C_{ep}[f_e,f_p],\\
    \frac{d f_p}{d\lambda}&=C_{p\gamma}[f_p,f_\gamma]+C_{pe}[f_p,f_e],\\
    \frac{d f_\gamma}{d\lambda}&=C_{\gamma e}[f_\gamma,f_e]+C_{\gamma p}[f_\gamma,f_p].
\end{aligned}
\end{align}
Coulomb scattering is extremely efficient at keeping electrons and protons tightly coupled at \emph{all times}, allowing to treat them as a single baryon fluid with common bulk quantities to extreme accuracy
\begin{align}
    \delta_e=\delta_p=\delta_b, \quad \vec{u}_e=\vec{u}_p=\vec{u}_b.
\end{align}
The second interaction keeps photons tightly coupled to electrons, and thus to protons, making sense of the photon-baryon fluid picture at early times.
The last interaction is negligible since the Thompson cross-section $\sigma_T \propto m^{-2}$ and thus for protons is suppressed by a factor $(m_e/m_p)^2$.

Thompson scattering does not change number density, but does exchange momentum between photons and baryons, and thus appears as a source term in the Euler equations of the two fluids, with opposite signs to ensure total momentum conservation.
That is, the conservation equations read
\begin{align}
    \nabla_\mu T^\mu_{(\gamma)\,0}=\nabla_\mu T^\mu_{(b)\,0}=0\quad \text{while}\quad \nabla_\mu T^\mu_{(\gamma)\,j}=-\nabla_\mu T^\mu_{(b)\,j}=:Q_j\neq 0.
\end{align}
Regardless of the interaction, the RHS for the Euler equations is written in terms of the velocity $v_j=\frac{1}{(\rho+p)}T^0_{\;j}$.
Therefore if we know the RHS of either the photon or baryon Euler equation, we can immediately get the RHS of the other by
{\small
\begin{align}
    \frac{\partial v_\gamma^j}{\partial \tau}+\ldots = \frac{1}{(\rho_\gamma+p_\gamma)} Q_j=:X\quad\Rightarrow\quad \frac{\partial v_b^j}{\partial \tau}+\ldots = -\frac{1}{(\rho_b+p_b)} Q_j = -\frac{(\rho_\gamma+p_\gamma)}{(\rho_b+p_b)} \frac{1}{(\rho_\gamma+p_\gamma)} Q_j = -\frac{4}{3} \frac{\rho_\gamma}{\rho_b} X.
\end{align}}

\section{CMB Power spectrum}
%=========================================================
%------------------------------------------------------------



We expand $\Theta(\vec{x}, \hat{p}, t)$ in spherical harmonics and Fourier components
{\small\begin{align}
\begin{aligned}
    a_{\ell m}(\vec{x}, t)=\int d \Omega_{\hat{p}}\, Y_{\ell m}^{*}(\hat{p}) \Theta(\vec{x}, \hat{p}, t) )=\int \frac{d^{3} k}{(2 \pi)^{3}} e^{i \vec{k} \cdot \vec{x}} \int d \Omega_{\hat{p}} Y_{\ell m}^{*}(\hat{p}) \Theta(\vec{k}, \hat{p}, t).
\end{aligned}
\end{align}}
Inflation predicts that the initial perturbations (like $\mathcal{R}$) form a random field that is very close to Gaussian.
Consequently $\Theta$ and the $a_{\ell m}$ will also form a Gaussian random field with mean zero and some variance $C_{\ell}$, i.e.
{\small\begin{align}
\begin{aligned}
    \langle a_{\ell m} a^{*}_{\ell' m'}\rangle = \delta_{\ell\ell'}\delta_{mm'} C_{\ell}
\end{aligned}
\end{align}}
where the brackets denote an ensemble average (average over many different realizations of our Universes).
And since it is Gaussian all the statistical information about $\Theta$ is contained in $C_{\ell}$ (in practice there will be some deviations from pure Gaussianity depending on the model of inflation so higher order moments do potentially contain interesting information).
For a fixed $\ell$, each $a_{\ell m}$ has the same variance and for each value of $\ell$ there are $2\ell+1$ possible values of $m$.
Thus in an experiment after measuring the $a_{\ell m}$'s we can estimate the power-spectrum as
{\small\begin{align}
\begin{aligned}
    \hat{C}_{\ell}=\frac{1}{2\ell+1}\sum_{m=-\ell}^{\ell} |a_{\ell m}|^{2}
\end{aligned}
\end{align}}
There is an important difference with what we measure and what the theory says.
The theory is formulated in terms of an ensemble average (average of many Universes) whereas we only have one Universe to measure in.
This is where the ergodic assumption comes into play, which tells us that as long as we have enough modes the mean of these will be the same as the ensemble average.
This means that for large $\ell$'s we have enough statistical power to accurately estimate $C_{\ell}$, while for very low $\ell$ we will run into cosmic variance: we have very few $m$'s to estimate $C_{\ell}$ from so there is a fundamental uncertainty that is due to us only having one single Universe to measure this in that we cannot get around.
Since $\hat{C}_\ell$ is the sum of squared Gaussians it follows a $\chi^2$ distribution with uncertainty given by
{\small\begin{align}
\begin{aligned}
    \frac{\mathrm{Var}(\hat{C}_{\ell})}{C_{\ell}^{2}}=\frac{2}{2\ell+1}, \quad\text{with}\quad \hat{C}_{\ell}=\frac{1}{2\ell+1}\sum_{m}|a_{\ell m}|^{2} \sim \,\,\chi^2 \text{ distributed}
\end{aligned}
\end{align}}
This uncertainty is what is depicted by the green shaded region (and the large red error-bars) for low $\ell$ in Figure 1 and shows that our current measurements for temperature fluctuations are close to as good as we can get.

To derive an expression for $C_{\ell}$ we expand $\Theta$ in multipoles
{\small\begin{align}
\begin{aligned}
    a_{\ell m}(\vec{x}, t)=\sum_{\ell_{1}} (2\ell_{1}+1)(-i)^{\ell_{1}} \int \frac{d^{3}k}{(2\pi)^{3}} e^{i\vec{k}\cdot \vec{x}} \int d\Omega_{\hat{p}}\, Y_{\ell m}^{*}(\hat{p}) P_{\ell_{1}}(\mu)\Theta_{\ell_{1}}(\vec{k}, t)
\end{aligned}
\end{align}}
where $\mu=\hat{p}\cdot \hat{k}$ which gives us
{\small\begin{align}
\begin{aligned}
    a_{\ell m}a_{\ell' m'}^{*}=\sum_{\ell_{1},\ell_{2}} &(2\ell_{1}+1)(2\ell_{2}+1)(-i)^{\ell_{1}-\ell_{2}} \int \frac{d^{3}k\, d^{3}k'}{(2\pi)^{6}} e^{i(\vec{k}-\vec{k}')\cdot \vec{x}}\\
    &\int d\Omega_{\hat{p}} d\Omega_{\hat{p}'}Y_{\ell m}^{*}(\hat{p})Y_{\ell' m'}(\hat{p}')P_{\ell_{1}}(\mu)P_{\ell_{2}}(\mu')\Theta_{\ell_{1}}(\vec{k}, t)\Theta_{\ell_{2}}^{*}(\vec{k}', t).
\end{aligned}
\end{align}}
Now remember that when we solve this numerically we put $\Theta_{\ell}(\vec{k}, t)=\Theta_{\ell}^{\mathrm{code}}(k, t)\mathcal{R}_{\mathrm{ini}}(\vec{k})$ where the initial value of the curvature perturbation $\mathcal{R}_{\mathrm{ini}}\propto \Psi$ was set to unity, i.e. we factored out the common initial condition from inflation before solving (which is perfectly fine since the equation system is linear).
We must now put it back in.
When we take the ensemble average of the above this will lead to a term
{\small\begin{align}
\begin{aligned}
    \langle \mathcal{R}_{\mathrm{ini}}(\vec{k})\mathcal{R}_{\mathrm{ini}}^{*}(\vec{k}')\rangle =(2\pi)^{3}\delta(\vec{k}-\vec{k}')P(k)
\end{aligned}
\end{align}}
where the primordial power-spectrum is given by
{\small\begin{align}
\begin{aligned}
    P(k)=\frac{2\pi^{2}}{k^{3}}\mathcal{P}_{\mathcal{R}}(k), \qquad \mathcal{P}_{\mathcal{R}}(k)=A_{s}(k/k_{\mathrm{pivot}})^{n_{s}-1}\,.
\end{aligned}
\end{align}}
The delta-function will enforce $\vec{k}=\vec{k}'$ taking care of one of the integrals and
{\small\begin{align}
\begin{aligned}
    \langle a_{\ell m}a_{\ell' m'}^{*}\rangle =\sum_{\ell_{1},\ell_{2}}& (2\ell_{1}+1)(2\ell_{2}+1)(-i)^{\ell_{1}-\ell_{2}} \int \frac{d^{3}k}{(2\pi)^{3}} \int d\Omega_{\hat{p}} d\Omega_{\hat{p}'}\\
    &\cdot Y_{\ell m}^{*}(\hat{p})Y_{\ell' m'}(\hat{p}')P_{\ell_{1}}(\mu)P_{\ell_{2}}(\mu')\Theta_{\ell_{1}}^{\mathrm{code}}(k, t)\Theta_{\ell_{2}}^{\mathrm{code}*}(k, t)\frac{2\pi^{2}\mathcal{P}_{\mathcal{R}}(k)}{k^{3}}
\end{aligned}
\end{align}}
For the next step we need
{\small\begin{align}
\begin{aligned}
    \int d\Omega_{\hat{p}}\, P_{\ell}(\mu)Y_{\ell' m}^{*}(\hat{p})=\delta_{\ell\ell'}\frac{4\pi}{2\ell+1}Y_{\ell m}^{*}(\hat{k})
\end{aligned}
\end{align}}
which takes care of two of the integrals (and the two Kronecker deltas kill both of the sums) leaving us with
{\small\begin{align}
\begin{aligned}
    \langle a_{\ell m}a_{\ell' m'}^{*}\rangle =\delta_{\ell\ell'}(4\pi)^{2}\int \frac{d^{3}k}{(2\pi)^{3}}\, Y_{\ell m}(\hat{k})Y_{\ell m'}^{*}(\hat{k})|\Theta_{\ell}^{\mathrm{code}}(k, t)|^{2}\frac{2\pi^{2}\mathcal{P}_{\mathcal{R}}(k)}{k^{3}}
\end{aligned}
\end{align}}
Finally splitting the $k$-integral into an integral over angles ($\Omega_{\hat{k}}$ now means over the angles of $k$) using that
{\small\begin{align}
\begin{aligned}
    \int d\Omega_{\hat{k}}\, Y_{\ell m}(\hat{k})Y_{\ell m'}^{*}(\hat{k})=\delta_{mm'}
\end{aligned}
\end{align}}
we get
{\small\begin{align}
\begin{aligned}
    \langle a_{\ell m}a_{\ell' m'}^{*}\rangle =\delta_{\ell\ell'}\delta_{mm'}4\pi \int \frac{dk}{k}\,\mathcal{P}_{\mathcal{R}}(k)|\Theta_{\ell}^{\mathrm{code}}(k, t)|^{2}
\end{aligned}
\end{align}}
so
{\small\begin{align}
\begin{aligned}
    C_{\ell}=4\pi \int \mathcal{P}_{\mathcal{R}}(k)|\Theta_{\ell}^{\mathrm{code}}(k)|^{2}\frac{dk}{k}
\end{aligned}
\end{align}}
where the time is taken to be today (which is the time when we observe the CMB, so $\Theta_{\ell}^{\mathrm{code}}(k)=\Theta_{\ell}^{\mathrm{code}}(k, t=t_{\mathrm{today}})$).
The difference with respect to what is in Chapter 8.5.2 in Dodelson is that he expresses this in terms of
{\small\begin{align}
\begin{aligned}
    P(k)=\frac{2\pi^{2}}{k^{3}}\mathcal{P}_{\mathcal{R}}(k)
\end{aligned}
\end{align}}
which gives us the equivalent formulation
{\small\begin{align}
\begin{aligned}
    C_{\ell}=\frac{2}{\pi}\int k^{2}P(k)|\Theta_{\ell}^{\mathrm{code}}(k)|^{2}dk
\end{aligned}
\end{align}}
The expression above tells us how to go from the stuff we computed numerically, the temperature multipoles $\Theta_{\ell}$, to the theoretical prediction for the CMB angular power-spectrum.
This can be directly compared to the equivalent quantity measured in experiments.
